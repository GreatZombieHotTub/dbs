SQL> CREATE OR REPLACE TRIGGER trg_validate_instructor
  2  BEFORE INSERT OR UPDATE ON instructor
  3  FOR EACH ROW
  4  DECLARE
  5      dept_budget department.budget%TYPE;
  6  BEGIN
  7      -- Name validation (only alphabets)
  8      IF NOT REGEXP_LIKE(:NEW.name, '^[A-Za-z ]+$') THEN
  9          RAISE_APPLICATION_ERROR(-20001, 'Name must contain only alphabets');
 10      END IF;
 11
 12      -- Salary > 0
 13      IF :NEW.salary <= 0 THEN
 14          RAISE_APPLICATION_ERROR(-20002, 'Salary must be positive');
 15      END IF;
 16
 17      -- Check department budget
 18      SELECT budget INTO dept_budget
 19      FROM department
 20      WHERE dept_name = :NEW.dept_name;
 21
 22      IF :NEW.salary > dept_budget THEN
 23          RAISE_APPLICATION_ERROR(-20003, 'Salary exceeds department budget');
 24      END IF;
 25
 26  EXCEPTION
 27      WHEN NO_DATA_FOUND THEN
 28          RAISE_APPLICATION_ERROR(-20004, 'Department not found');
 29  END;
